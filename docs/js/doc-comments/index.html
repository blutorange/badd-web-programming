/**
 * Checks if a `value` starts with a given `prefix`. __Both values must be
 * strings.__
 * 
 * If the prefix is the [empty string](https://en.wikipedia.org/wiki/Empty_string),
 * always returns `true`.
 * 
 * If the value is the empty string, always returns `false`, unless the prefix
 * is also the empty string.
 * 
 * @example
 * ```js
 * startsWith("foo-bar", "foo-") // => true
 * startsWith("foo-bar", "-bar") // => false
 * startsWith("foo", "") // => true
 * startsWith("", "foo") // => false
 * startsWith("", "")    // => true
 * ```
 * 
 * @param {string} value - A string to check whether it starts with a prefix,
 * e.g. `foo-bar`.
 * @param {string} prefix - A prefix to test, e.g. `foo-`.
 * @returns {boolean} `true` if the value starts with the given prefix, `false`
 * otherwise.
 */
function startsWith(value, prefix) {
  if (prefix.length === 0) {
    return true;
  }
  if (value.length === 0) {
    return false;
  }
  return value.substring(0, prefix.length) === prefix;
}
